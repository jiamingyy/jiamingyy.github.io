C语言函数指针语法

c
// 待用函数指针形参的函数声明
UG_S16 UG_Init(UG_GUI* g, void (*p)(UG_S16, UG_S16, UG_COLOR), UG_S16 x, UG_S16 y){
	// 强制类型转换
    p_temp = (void (*)(UG_S16, UG_S16, UG_COLOR)) p;
    // 可以在内部直接调用
    p(x, y, z);
    
}
